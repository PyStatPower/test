 38def fun_power(
 39    size: float,
 40    alpha: float,
 41    nullproportion: float,
 42    proportion: float,
 43    alternative: Alternative,
 44    test_type: TestType,
 45) -> float:
 46    """计算检验效能。
 47
 48    Parameters
 49    ----------
 50        size : int
 51            样本量
 52        alpha : float
 53            显著性水平
 54        nullproportion : float
 55            零假设的概率
 56        proportion : float
 57            备择假设的概率
 58        alternative : Alternative
 59            检验方法，可选值为 `Alternative.ONE_SIDED`, `Alternative.TWO_SIDED`
 60        test_type : TestType
 61            检验类型，可选值为 `TestType.EXACT_TEST`, `TestType.Z_TEST_USING_S_P0`,
 62                               `TestType.Z_TEST_USING_S_P0_CC`, `TestType.Z_TEST_USING_S_PHAT`,
 63                               `TestType.Z_TEST_USING_S_PHAT_CC`
 64    Returns
 65    -------
 66        power : float
 67            检验效能
 68    """
 69
 70    p0 = nullproportion
 71    p1 = proportion
 72
 73    denominator = sqrt(p1 * (1 - p1))
 74
 75    gMean = sqrt(size) * (p0 - p1) / denominator
 76
 77    gsd = sqrt(p0 * (1 - p0)) / denominator
 78
 79    # 使用 S(Phat) 作为标准差
 80    if test_type in (TestType.Z_TEST_USING_S_PHAT, TestType.Z_TEST_USING_S_PHAT_CC):
 81        gsd = sqrt(p1 * (1 - p1)) / denominator
 82
 83    # 连续性校正
 84    c = 0
 85    if test_type in (TestType.Z_TEST_USING_S_P0_CC, TestType.Z_TEST_USING_S_PHAT_CC):
 86        if abs(p1 - p0) > 1 / (2 * size):
 87            c = 1 / (2 * sqrt(size))
 88    gc = c / denominator
 89
 90    # 计算检验效能
 91    if alternative == Alternative.ONE_SIDED:
 92        if p1 < p0:
 93            z = norm.ppf(1 - alpha)
 94            stat = gMean - gsd * z - gc
 95            result = norm.cdf(stat)
 96        elif p1 > p0:
 97            z = norm.ppf(1 - alpha)
 98            stat = gMean + gsd * z + gc
 99            result = 1 - norm.cdf(stat)
100    elif alternative == Alternative.TWO_SIDED:
101        z = norm.ppf(1 - alpha / 2)
102        stat = [gMean - gsd * z - gc, gMean + gsd * z + gc]
103        result = norm.cdf(stat[0]) + 1 - norm.cdf(stat[1])
104
105    return result

110        def __init__(
111            self,
112            alpha: Alpha,
113            power: Power,
114            nullproportion: Proportion,
115            proportion: Proportion,
116            alternative: Alternative,
117            test_type: TestType,
118            dropout_rate: DropOutRate,
119        ):
120            self.alpha = alpha
121            self.power = power
122            self.nullproportion = nullproportion
123            self.proportion = proportion
124            self.alternative = alternative
125            self.test_type = test_type
126            self.dropout_rate = dropout_rate
127
128        def solve(self) -> Size:
129            self._eval = (
130                lambda size: fun_power(
131                    size,
132                    alpha=self.alpha,
133                    nullproportion=self.nullproportion,
134                    proportion=self.proportion,
135                    alternative=self.alternative,
136                    test_type=self.test_type,
137                )
138                - self.power
139            )
140
141            lbound, ubound = Size.pseudo_bound()
142            try:
143                size = brentq(self._eval, lbound, ubound)
144            except ValueError as e:
145                raise ValueError("无解") from e
146
147            self.size = Size(ceil(size))
148
149            self.size_include_dropouts = Size(ceil(self.size / (1 - self.dropout_rate)))
150            self.dropouts = DropOutSize(self.size_include_dropouts - self.size)

240        def solve(self) -> Proportion:
241            self._eval = (
242                lambda nullproportion: fun_power(
243                    self.size, self.alpha, nullproportion, self.proportion, self.alternative, self.test_type
244                )
245                - self.power
246            )
247
248            match self.search_direction:
249                case SearchDirection.LESS:
250                    lbound, ubound = Interval(0, self.proportion).pseudo_bound()

261                case _:  # pragma: no cover

320def solve_for_sample_size(
321    alpha: float,
322    power: float,
323    nullproportion: float,
324    proportion: float,
325    alternative: str,
326    test_type: str,
327    dropout_rate: float = 0,
328    full_output: bool = False,
329):
330    """Calculate the sample size for one proportion test.
331
332    Parameters
333    ----------
334    alpha : float
335        Significance level.
336    power : float
337        Power of the test.
338    nullproportion : float
339        The proportion under the null hypothesis.
340    proportion : float
341        The proportion under the alternative hypothesis.
342    alternative : {"TWO_SIDED", "ONE_SIDED"}
343        The type of alternative hypothesis.
344    test_type : {"EXACT_TEST", "Z_TEST_USING_S_P0", "Z_TEST_USING_S_P0_CC", "Z_TEST_USING_S_PHAT", "Z_TEST_USING_S_PHAT_CC"}
345        The type of statistical test.
346    dropout_rate : float, default=0
347        The dropout rate.
348    full_output : bool, default=False
349        Whether to return the full output.
350
351    Returns
352    -------
353    Size | OneProportion.ForSize
354        The sample size for one proportion test. If `full_output` is True, return a `OneProportion.ForSize` object.
355    """
356
357    model = OneProportion.ForSize(
358        alpha=Alpha(alpha),
359        power=Power(power),
360        nullproportion=Proportion(nullproportion),
361        proportion=Proportion(proportion),
362        alternative=Alternative[alternative],
363        test_type=TestType[test_type],
364        dropout_rate=DropOutRate(dropout_rate),
365    )
366    model.solve()
367
368    if full_output:
369        return model
370    return model.size
