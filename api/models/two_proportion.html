422def fun_power(
423    alpha: float,
424    treatment_n: float,
425    reference_n: float,
426    treatment_proportion: float,
427    reference_proportion: float,
428    alternative: Alternative,
429    test_type: TestType,
430):
431    """_summary_
432
433    Args:
434        alpha (float): 显著性水平
435        treatment_n (float): 试验组样本量
436        reference_n (float): 对照组样本量
437        treatment_proportion (float): 试验组的率
438        reference_proportion (float): 对照组的率
439        alternative (Alternative): 备择假设类型
440        test_type (TestType): 检验类型
441
442    Returns:
443        power (float): 检验效能
444    """
445    n1 = treatment_n
446    n2 = reference_n
447    p1 = treatment_proportion
448    p2 = reference_proportion
449    q1 = 1 - p1
450    q2 = 1 - p2
451
452    p_bar = (n1 * p1 + n2 * p2) / (n1 + n2)
453    q_bar = 1 - p_bar
454
455    se_p = sqrt(p_bar * q_bar * (1 / n1 + 1 / n2))
456    se_u = sqrt(p1 * q1 / n1 + p2 * q2 / n2)
457
458    # 计算标准误
459    match test_type:
460        case TestType.Z_TEST_POOLED | TestType.Z_TEST_CC_POOLED:
461            se = se_p
462        case TestType.Z_TEST_UNPOOLED | TestType.Z_TEST_CC_UNPOOLED:
463            se = se_u
464        case _:  # pragma: no cover
465            assert False, "未知的检验类型"
466
467    # 连续性校正
468    c = 0
469    if test_type in [TestType.Z_TEST_CC_POOLED, TestType.Z_TEST_CC_UNPOOLED]:
470        c = (1 / 2) * (1 / n1 + 1 / n2)
471
472    # 计算检验效能
473    match alternative:
474        case Alternative.TWO_SIDED:
475            z_alpha = norm.ppf(1 - alpha / 2)
476            power = (
477                norm.cdf((-z_alpha * se - c - (p1 - p2)) / se_u) + 1 - norm.cdf((z_alpha * se + c - (p1 - p2)) / se_u)
478            )
479        case Alternative.ONE_SIDED:
480            z_alpha = norm.ppf(1 - alpha)
481            if p1 > p2:
482                power = 1 - norm.cdf((z_alpha * se + c - (p1 - p2)) / se_u)
483            elif p1 <= p2:
484                power = norm.cdf((-z_alpha * se - c - (p1 - p2)) / se_u)
485        case _:  # pragma: no cover
486            assert False, "未知的备择假设类型"
487
488    return power

494    class ForSize:

497        def __init__(
498            self,
499            alpha: Alpha,
500            power: Power,
501            treatment_proportion: Proportion,

517        def solve(self):
