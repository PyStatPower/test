 11@dataclass(frozen=True)
 12class Interval:
 13    """定义一个数值区间，不支持单点区间，例如：[1, 1]。
 14
 15    Parameters
 16    ----------
 17    lower : int | float
 18        区间的下界。
 19    upper : int | float
 20        区间的上界。
 21    lower_inclusive : bool
 22        `True` 表示包含下界，`False` 表示不包含下界。默认为 `False`。
 23    upper_inclusive : bool
 24        `True` 表示包含上界，`False` 表示不包含上界。默认为 `False`。
 25
 26    Examples
 27    --------
 28    >>> itv= Interval(0, 1, lower_inclusive=True, upper_inclusive=False)
 29    >>> itv
 30    [0, 1)
 31    >>> 0.5 in itv
 32    True
 33    >>> 1 in itv
 34    False
 35    >>> 0 in itv
 36    False
 37    >>> itv.pseudo_bound()
 38    (0, 0.9999999999)
 39    """
 40
 41    lower: int | float
 42    upper: int | float
 43    lower_inclusive: bool = False
 44    upper_inclusive: bool = False
 45
 46    def __contains__(self, value: int | float) -> bool:
 47        if isinstance(value, (int, float)):
 48            if self.lower_inclusive:
 49                if self.upper_inclusive:
 50                    return self.lower <= value <= self.upper
 51                else:
 52                    return self.lower <= value < self.upper
 53            else:
 54                if self.upper_inclusive:
 55                    return self.lower < value <= self.upper
 56                else:
 57                    return self.lower < value < self.upper
 58
 59        raise TypeError(f"Interval.__contains__ only supports real numbers, but you passed in a {type(value)}.")
 60
 61    def __eq__(self, other: object) -> bool:
 62        if isinstance(other, Interval):
 63            return (
 64                isclose(self.lower, other.lower)
 65                and isclose(self.upper, other.upper)
 66                and self.lower_inclusive == other.lower_inclusive
 67                and self.upper_inclusive == other.upper_inclusive
 68            )
 69
 70        raise TypeError(f"Interval.__eq__ only supports Interval, but you passed in a {type(other)}.")
 71
 72    def __repr__(self) -> str:
 73        if self.lower_inclusive:
 74            if self.upper_inclusive:
 75                return f"[{self.lower}, {self.upper}]"
 76            else:
 77                return f"[{self.lower}, {self.upper})"
 78        else:
 79            if self.upper_inclusive:
 80                return f"({self.lower}, {self.upper}]"
 81            else:
 82                return f"({self.lower}, {self.upper})"

 84    def pseudo_lbound(self, eps: float = MIN_FLOAT) -> int | float:
 85        """返回区间的伪下界，用于数值计算。
 86
 87        Parameters
 88        ----------
 89        eps : float
 90            用于计算伪下界的 ε 。默认为 :const:`MIN_FLOAT`。
 91
 92        Returns
 93        -------
 94        int | float
 95            区间的伪下界。
 96        """
 97
 98        if self.lower_inclusive:
 99            return self.lower
100        else:
101            return self.lower + eps

139class PowerAnalysisFloat(float):
140    """功效分析数值类型的基类。
141
142    - 如果传递一个 `int` 或 `float` 数值来创建一个实例，将会检查它是否在定义域 :attr:`domain` 内。
143      如果不在，将会引发一个 `ValueError`，否则将返回一个新的浮点数对象，其行为与内置的浮点数对象相同。
144    - 如果传递 `None` 来创建一个实例，将会返回 `None`。
145    - 如果传递其他类型来创建一个实例，将会引发一个 `TypeError`。
146
147    Examples
148    --------
149    >>> PowerAnalysisFloat(0.5)
150    0.5
151    >>> PowerAnalysisFloat(0.5) * PowerAnalysisFloat(0.5)
152    0.25
153    >>> isinstance(PowerAnalysisFloat(0.5), float)
154    True
155    """
156
157    #: 一个 :class:`Interval` 对象，用于限制特定数值类型的取值范围。
158    domain = Interval(-MAX_FLOAT, MAX_FLOAT, lower_inclusive=True, upper_inclusive=True)
159
160    def __new__(cls, obj):
161        if isinstance(obj, (int, float)):
162            if obj not in cls.domain:
163                raise ValueError(f"{obj} is not in {cls.domain}.")
164            return super().__new__(cls, obj)
165        elif obj is None:
166            return None
167        else:
168            raise TypeError(f"{obj} must be either an int, float, or None.")
