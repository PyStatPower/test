 4class Option(EnumMeta):
 5    """
 6    功效分析选项的枚举元类，可支持通过大小写不敏感的枚举值对枚举成员进行访问。
 7
 8    Examples
 9    --------
10    >>> class DayOfWeek(Enum, metaclass=Option):
11    ...     SUN = 1
12    ...     MON = 2
13    ...     TUE = 3
14    ...
15    >>> DayOfWeek["sun"]
16    <DayOfWeek.SUN: 1>
17    >>> DayOfWeek["Mon"]
18    <DayOfWeek.MON: 2>
19    >>> DayOfWeek["TUE"]
20    <DayOfWeek.TUE: 3>
21    """
22
23    def __getitem__(self, name):
24        if isinstance(name, str):
25            return super().__getitem__(name.upper())
26        else:
27            return super().__getitem__(name)
